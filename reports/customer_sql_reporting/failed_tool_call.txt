         function updateSankeyChart(data) {                                                                                                                                                                
             const sankeyData = data.journey_sankey;                                                                                                                                                       
             if (!sankeyData || !sankeyData.rows || sankeyData.rows.length === 0) {                                                                                                                        
                 document.getElementById('sankey-chart').innerHTML = '<div class="no-data">No journey flow data available</div>';                                                                          
                 return;                                                                                                                                                                                   
             }                                                                                                                                                                                             
                                                                                                                                                                                                           
             // Process Sankey data                                                                                                                                                                        
             const rows = sankeyData.rows;                                                                                                                                                                 
             const columns = sankeyData.columns;                                                                                                                                                           
                                                                                                                                                                                                           
             // Extract source, target, value                                                                                                                                                              
             const sourceIndex = columns.indexOf('source');                                                                                                                                                
             const targetIndex = columns.indexOf('target');                                                                                                                                                
             const valueIndex = columns.indexOf('value');                                                                                                                                                  
                                                                                                                                                                                                           
             const sources = [];                                                                                                                                                                           
             const targets = [];                                                                                                                                                                           
             const values = [];                                                                                                                                                                            
             const labels = [];                                                                                                                                                                            
                                                                                                                                                                                                           
             // Build unique node list                                                                                                                                                                     
             const nodeSet = new Set();                                                                                                                                                                    
             rows.forEach(row => {                                                                                                                                                                         
                 nodeSet.add(row[sourceIndex]);                                                                                                                                                            
                 nodeSet.add(row[targetIndex]);                                                                                                                                                            
             });                                                                                                                                                                                           
             const nodes = Array.from(nodeSet);                                                                                                                                                            
                                                                                                                                                                                                           
             // Map nodes to indices                                                                                                                                                                       
             const nodeMap = {};                                                                                                                                                                           
             nodes.forEach((node, index) => {                                                                                                                                                              
                 nodeMap[node] = index;                                                                                                                                                                    
             });                                                                                                                                                                                           
                                                                                                                                                                                                           
             // Build links                                                                                                                                                                                
             rows.forEach(row => {                                                                                                                                                                         
                 sources.push(nodeMap[row[sourceIndex]]);                                                                                                                                                  
                 targets.push(nodeMap[row[targetIndex]]);                                                                                                                                                  
                 values.push(row[valueIndex]);                                                                                                                                                             
                 labels.push(row[columns.indexOf('label')] || '');                                                                                                                                         
             });                                                                                                                                                                                           
                                                                                                                                                                                                           
             const trace = {                                                                                                                                                                               
                 type: 'sankey',                                                                                                                                                                           
                 orientation: 'h',                                                                                                                                                                         
                 node: {                                                                                                                                                                                   
                     pad: 15,                                                                                                                                                                              
                     thickness: 20,                                                                                                                                                                        
                     line: {                                                                                                                                                                               
                         color: 'black',                                                                                                                                                                   
                         width: 0.5                                                                                                                                                                        
                     },                                                                                                                                                                                    
                     label: nodes,                                                                                                                                                                         
                     color: nodes.map((_, i) => getColorForNode(i, nodes.length))                                                                                                                          
                 },                                                                                                                                                                                        
                 link: {                                                                                                                                                                                   
                     source: sources,                                                                                                                                                                      
                     target: targets,                                                                                                                                                                      
                     value: values,                                                                                                                                                                        
                     label: labels,                                                                                                                                                                        
                     color: sources.map(i => `rgba(${getColorForNode(i, nodes.length)}, 0.3)`)                                                                                                             
                 }                                                                                                                                                                                         
             };                                                                                                                                                                                            
                                                                                                                                                                                                           
             const layout = {                                                                                                                                                                              
                 title: 'Customer Journey Flow',                                                                                                                                                           
                 font: { size: 12 },                                                                                                                                                                       
                 margin: { t: 40, l: 50, r: 50, b: 40 }                                                                                                                                                    
             };                                                                                                                                                                                            
                                                                                                                                                                                                           
             const plotDiv = document.getElementById('sankey-chart');                                                                                                                                      
             Plotly.newPlot(plotDiv, [trace], layout, { responsive: true });                                                                                                                               
             reportState.sankeyChart = plotDiv;                                                                                                                                                            
         }                                                                                                                                                                                                 
                                                                                                                                                                                                           
         function updateSunburstChart(data) {                                                                                                                                                              
             const sunburstData = data.sunburst_hierarchy;                                                                                                                                                 
             if (!sunburstData || !sunburstData.rows || sunburstData.rows.length === 0) {                                                                                                                  
                 document.getElementById('sunburst-chart').innerHTML = '<div class="no-data">No hierarchical data available</div>';                                                                        
                 return;                                                                                                                                                                                   
             }                                                                                                                                                                                             
                                                                                                                                                                                                           
             const rows = sunburstData.rows;                                                                                                                                                               
             const columns = sunburstData.columns;                                                                                                                                                         
                                                                                                                                                                                                           
             const level1Index = columns.indexOf('level1');                                                                                                                                                
             const level2Index = columns.indexOf('level2');                                                                                                                                                
             const level3Index = columns.indexOf('level3');                                                                                                                                                
             const level4Index = columns.indexOf('level4');                                                                                                                                                
             const valueIndex = columns.indexOf('value');                                                                                                                                                  
             const pathIndex = columns.indexOf('hierarchy_path');                                                                                                                                          
                                                                                                                                                                                                           
             const labels = [];                                                                                                                                                                            
             const parents = [];                                                                                                                                                                           
             const values = [];                                                                                                                                                                            
                                                                                                                                                                                                           
             rows.forEach(row => {                                                                                                                                                                         
                 const level1 = row[level1Index];                                                                                                                                                          
                 const level2 = row[level2Index];                                                                                                                                                          
                 const level3 = row[level3Index];                                                                                                                                                          
                 const level4 = row[level4Index];                                                                                                                                                          
                 const value = row[valueIndex];                                                                                                                                                            
                                                                                                                                                                                                           
                 // Add all levels to hierarchy                                                                                                                                                            
                 if (level1) {                                                                                                                                                                             
                     labels.push(level1);                                                                                                                                                                  
                     parents.push('');                                                                                                                                                                     
                     values.push(null); // Root values calculated automatically                                                                                                                            
                 }                                                                                                                                                                                         
                                                                                                                                                                                                           
                 if (level2) {                                                                                                                                                                             
                     labels.push(level2);                                                                                                                                                                  
                     parents.push(level1);                                                                                                                                                                 
                     values.push(null);                                                                                                                                                                    
                 }                                                                                                                                                                                         
                                                                                                                                                                                                           
                 if (level3) {                                                                                                                                                                             
                     labels.push(level3);                                                                                                                                                                  
                     parents.push(level2);                                                                                                                                                                 
                     values.push(null);                                                                                                                                                                    
                 }                                                                                                                                                                                         
                                                                                                                                                                                                           
                 if (level4) {                                                                                                                                                                             
                     labels.push(level4);                                                                                                                                                                  
                     parents.push(level3);                                                                                                                                                                 
                     values.push(value); // Leaf nodes have actual values                                                                                                                                  
                 }                                                                                                                                                                                         
             });                                                                                                                                                                                           
                                                                                                                                                                                                           
             // Remove duplicates                                                                                                                                                                          
             const uniqueData = {};                                                                                                                                                                        
             labels.forEach((label, i) => {                                                                                                                                                                
                 const key = label + '|' + parents[i];                                                                                                                                                     
                 if (!uniqueData[key]) {                                                                                                                                                                   
                     uniqueData[key] = { label, parent: parents[i], value: values[i] };                                                                                                                    
                 }                                                                                                                                                                                         
             });                                                                                                                                                                                           
                                                                                                                                                                                                           
             const uniqueLabels = Object.values(uniqueData).map(d => d.label);                                                                                                                             
             const uniqueParents = Object.values(uniqueData).map(d => d.parent);                                                                                                                           
             const uniqueValues = Object.values(uniqueData).map(d => d.value);                                                                                                                             
                                                                                                                                                                                                           
             const trace = {                                                                                                                                                                               
                 type: 'sunburst',                                                                                                                                                                         
                 labels: uniqueLabels,                                                                                                                                                                     
                 parents: uniqueParents,                                                                                                                                                                   
                 values: uniqueValues,                                                                                                                                                                     
                 branchvalues: 'total',                                                                                                                                                                    
                 marker: {                                                                                                                                                                                 
                     colors: uniqueLabels.map((_, i) => getColorForNode(i, uniqueLabels.length))                                                                                                           
                 },                                                                                                                                                                                        
                 textinfo: 'label+value',                                                                                                                                                                  
                 hoverinfo: 'label+value+percent parent+percent root'                                                                                                                                      
             };                                                                                                                                                                                            
                                                                                                                                                                                                           
             const layout = {                                                                                                                                                                              
                 title: 'Journey Hierarchy',                                                                                                                                                               
                 margin: { t: 40, l: 0, r: 0, b: act}: 0 },                                                                                                                                                
                 sunburstcolorway: ['#636efa', '#EF553B', '#00cc96', '#ab63fa', '#FFA15A']                                                                                                                 
             };                                                                                                                                                                                            
                                                                                                                                                                                                           
             const plotDiv = document.getElementById('sunburst-chart');                                                                                                                                    
             Plotly.newPlot(plotDiv, [trace], layout, { responsive: true });                                                                                                                               
             reportState.sunburstChart = plotDiv;                                                                                                                                                          
         }                                                                                                                                                                                                 
                                                                                                                                                                                                           
         function updateParallelChart(data) {                                                                                                                                                              
             const parallelData = data.parallel_coordinates;                                                                                                                                               
             if (!parallelData || !parallelData.rows || parallelData.rows.length === 0) {                                                                                                                  
                 document.getElementById('parallel-chart').innerHTML = '<div class="no-data">No multi-dimensional data available</div>';                                                                   
                 return;                                                                                                                                                                                   
             }                                                                                                                                                                                             
                                                                                                                                                                                                           
             const rows = parallelData.rows;                                                                                                                                                               
             const columns = parallelData.columns;                                                                                                                                                         
                                                                                                                                                                                                           
             // Convert to array of objects                                                                                                                                                                
             const dimensions = [];                                                                                                                                                                        
             const dataArrays = [];                                                                                                                                                                        
                                                                                                                                                                                                           
             columns.forEach((col, colIndex) => {                                                                                                                                                          
                 if (col !== 'seeker_id') {                                                                                                                                                                
                     const values = rows.map(row => row[colIndex]);                                                                                                                                        
                     const range = [Math.min(...values), Math.max(...values)];                                                                                                                             
                                                                                                                                                                                                           
                     dimensions.push({                                                                                                                                                                     
                         label: col,                                                                                                                                                                       
                         values: values,                                                                                                                                                                   
                         range: range,                                                                                                                                                                     
                         tickformat: col.includes('count') ? 'd' : '.2f'                                                                                                                                   
                     });                                                                                                                                                                                   
                 }                                                                                                                                                                                         
             });                                                                                                                                                                                           
                                                                                                                                                                                                           
             // Create traces for each dimension                                                                                                                                                           
             const traces = dimensions.map((dim, i) => {                                                                                                                                                   
                 return {                                                                                                                                                                                  
                     type: 'parcoords',                                                                                                                                                                    
                     dimensions: dimensions.map(d => ({                                                                                                                                                    
                         label: d.label,                                                                                                                                                                   
                         values: d.values,                                                                                                                                                                 
                         range: d.range,                                                                                                                                                                   
                         tickformat: d.tickformat                                                                                                                                                          
                     })),                                                                                                                                                                                  
                     line: {                                                                                                                                                                               
                         color: rows.map((_, idx) => getColorForNode(idx, rows.length)),                                                                                                                   
                         colorscale: 'Viridis'                                                                                                                                                             
                     }                                                                                                                                                                                     
                 };                                                                                                                                                                                        
             });                                                                                                                                                                                           
                                                                                                                                                                                                           
             const layout = {                                                                                                                                                                              
                 title: 'Multi-dimensional Seeker Analysis',                                                                                                                                               
                 margin: { t: 40, l: 60, r: 30, b: 30 }                                                                                                                                                    
             };                                                                                                                                                                                            
                                                                                                                                                                                                           
             const plotDiv = document.getElementById('parallel-chart');                                                                                                                                    
             Plotly.newPlot(plotDiv, traces.slice(0,1), layout, { responsive: true });                                                                                                                     
             reportState.parallelChart = plotDiv;                                                                                                                                                          
         }                                                                                                                                                                                                 
                                                                                                                                                                                                           
         function update3DChart(data) {                                                                                                                                                                    
             const scatterData = data.scatter_3d;                                                                                                                                                          
             if (!scatterData || !scatterData.rows || scatterData.rows.length === 0) {                                                                                                                     
                 document.getElementById('3d-chart').innerHTML = '<div class="no-data">No 3D plot data available</div>';                                                                                   
                 return;                                                                                                                                                                                   
             }                                                                                                                                                                                             
                                                                                                                                                                                                           
             const rows = scatterData.rows;                                                                                                                                                                
             const columns = scatterData.columns;                                                                                                                                                          
                                                                                                                                                                                                           
             const dateIndex = columns.indexOf('assessment_date');                                                                                                                                         
             const complexityIndex = columns.indexOf('question_complexity');                                                                                                                               
             const outcomeIndex = columns.indexOf('completion_rate');                                                                                                                                      
             const diversityIndex = columns.indexOf('program_diversity');                                                                                                                                  
             const seekerIndex = columns.indexOf('seeker_count');                                                                                                                                          
                                                                                                                                                                                                           
             const dates = rows.map(row => new Date(row[dateIndex]).getTime());                                                                                                                            
             const complexities = rows.map(row => row[complexityIndex]);                                                                                                                                   
             const outcomes = rows.map(row => row[outcomeIndex]);                                                                                                                                          
             const diversities = rows.map(row => row[diversityIndex]);                                                                                                                                     
             const seekers = rows.map(row => row[seekerIndex]);                                                                                                                                            
                                                                                                                                                                                                           
             const trace = {                                                                                                                                                                               
                 type: 'scatter3d',                                                                                                                                                                        
                 mode: 'markers',                                                                                                                                                                          
                 x: dates,                                                                                                                                                                                 
                 y: complexities,
