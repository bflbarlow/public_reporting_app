
    <script>
        // SIMPLE, ROBUST DASHBOARD IMPLEMENTATION
        // This avoids complex error handling and template bug workarounds
        
        // Global state
        const state = {
            charts: {
                dailyTrends: null,
                statusDistribution: null,
                contactMethod: null,
                monthlySummary: null
            },
            data: {
                daily_trends: { rows: [] },
                status_distribution: { rows: [] },
                contact_method_analysis: { rows: [] },
                monthly_summary: { rows: [] }
            },
            filters: {}
        };

        // Initialize the dashboard
        function initDashboard() {
            console.log('Initializing dashboard...');
            
            // Setup error handling
            setupErrorHandling();
            
            // Initialize filter values from current URL parameters
            initFilters();
            
            // Setup event listeners
            setupEventListeners();
            
            // Load initial data
            loadData();
        }

        function setupErrorHandling() {
            // Prevent unhandled errors from breaking the dashboard
            window.addEventListener('error', function(event) {
                console.error('Dashboard error:', event.error || event.message);
                event.preventDefault();
            });
            
            window.addEventListener('unhandledrejection', function(event) {
                console.error('Unhandled promise rejection:', event.reason);
                event.preventDefault();
            });
        }

        function initFilters() {
            // Get current parameters from the report config
            const config = window.__REPORT_CONFIG__ || {};
            const params = config.params || {};
            
            // Set filter values
            if (params.start_date) {
                document.getElementById('start-date').value = params.start_date;
            }
            
            if (params.end_date) {
                document.getElementById('end-date').value = params.end_date;
            }
            
            // Initialize state.filters with current parameters
            state.filters = {
                start_date: params.start_date || '',
                end_date: params.end_date || '',
                referral_status: params.referral_status || '',
                contact_method: params.contact_method || ''
            };
            
            // Note: referral_status and contact_method are deprecated but kept in state
            // for compatibility with existing URLs that may contain these parameters
        }

        function updateFilterState() {
            state.filters = {
                start_date: document.getElementById('start-date').value,
                end_date: document.getElementById('end-date').value,
                referral_status: state.filters.referral_status || '',  // preserve existing value
                contact_method: state.filters.contact_method || ''     // preserve existing value
            };
        }

        function setupEventListeners() {
            // Apply filters button
            document.getElementById('apply-filters').addEventListener('click', () => {
                updateFilterState();
                loadData();
            });
            
            // Reset filters button
            document.getElementById('reset-filters').addEventListener('click', () => {
                // Reset to original parameters
                const config = window.__REPORT_CONFIG__ || {};
                const params = config.params || {};
                
                document.getElementById('start-date').value = params.start_date || '';
                document.getElementById('end-date').value = params.end_date || '';
                
                // Reset all filter values including deprecated ones
                state.filters = {
                    start_date: params.start_date || '',
                    end_date: params.end_date || '',
                    referral_status: params.referral_status || '',
                    contact_method: params.contact_method || ''
                };
                
                loadData();
            });
            
            // Enter key in filter inputs
            const filterInputs = document.querySelectorAll('.filters-container input, .filters-container select');
            filterInputs.forEach(input => {
                input.addEventListener('keypress', function(e) {
                    if (e.key === 'Enter') {
                        updateFilterState();
                        loadData();
                    }
                });
            });
        }

        async function loadData() {
            console.log('Loading data with filters:', state.filters);
            
            const loadingEl = document.getElementById('loading-overlay');
            const errorEl = document.getElementById('error-message');
            const applyBtn = document.getElementById('apply-filters');
            
            // Show loading state
            loadingEl.classList.add('active');
            applyBtn.disabled = true;
            errorEl.style.display = 'none';
            
            try {
                // Validate date range if both dates are set
                if (state.filters.start_date && state.filters.end_date) {
                    if (state.filters.start_date > state.filters.end_date) {
                        throw new Error('Start date must be before end date.');
                    }
                }
                
                // Ensure thick client is ready and has current URL
                if (!window.ReportApp || !window.ReportApp.refresh) {
                    throw new Error('Thick client not available. Ensure <script src="/static/thick_client.js"> is included.');
                }
                
                // Thick client ready, load data
                
                // Try to load data via thick client
                console.log('Using thick client to load data');
                const data = await window.ReportApp.refresh(state.filters);
                console.log('Data loaded via thick client:', Object.keys(data));
                
                // Update state with new data
                updateDataState(data);
                
                // Update the dashboard with new data
                updateDashboard();
                
            } catch (error) {
                console.error('Error loading data:', error);
                showError('Failed to load data: ' + error.message);
            } finally {
                // Hide loading state
                loadingEl.classList.remove('active');
                applyBtn.disabled = false;
            }
        }

        function updateDataState(newData) {
            console.log('updateDataState called with:', newData);
            // Update each datasource with new data or empty structure
            const datasources = ['daily_trends', 'status_distribution', 'contact_method_analysis', 'monthly_summary'];
            datasources.forEach(dsName => {
                if (newData && newData[dsName]) {
                    console.log(`Setting ${dsName} data:`, newData[dsName]);
                    console.log(`Rows for ${dsName}:`, newData[dsName].rows ? newData[dsName].rows.length : 'none');
                    if (newData[dsName].rows && newData[dsName].rows.length > 0) {
                        console.log(`First row for ${dsName}:`, newData[dsName].rows[0]);
                    }
                    state.data[dsName] = newData[dsName];
                } else {
                    console.log(`No data for ${dsName}, setting empty`);
                    state.data[dsName] = { rows: [] };
                }
            });
            console.log('state.data after update:', state.data);
        }

        function updateDashboard() {
            console.log('updateDashboard called');
            try {
                // Update charts
                updateCharts();
                
                // Update data table
                updateDataTable();
                
                // Update stats panel
                updateStatsPanel();
                
                // Display immutable parameters
                displayImmutableParams();
                
            } catch (error) {
                console.error('Error updating dashboard:', error);
                showError('Error updating dashboard: ' + error.message);
            }
        }



        function updateCharts() {
            console.log('updateCharts called');
            // Destroy existing charts
            destroyCharts();
            
            // Create new charts
            createDailyTrendsChart();
            createStatusDistributionChart();
            createContactMethodChart();
            createMonthlySummaryChart();
        }

        function createDailyTrendsChart() {
            console.log('createDailyTrendsChart called');
            const canvas = document.getElementById('dailyTrendsChart');
            if (!canvas) {
                console.error('Canvas dailyTrendsChart not found');
                return;
            }
            const ctx = canvas.getContext('2d');
            const data = state.data.daily_trends || { rows: [] };
            console.log('daily_trends data:', data);
            
            // Show/hide no-data message
            const container = canvas.parentElement;
            let noDataMsg = container.querySelector('.no-data-message');
            if (!noDataMsg) {
                noDataMsg = document.createElement('div');
                noDataMsg.className = 'no-data-message';
                noDataMsg.textContent = 'No data available';
                container.appendChild(noDataMsg);
            }
            
            if (!data.rows || data.rows.length === 0) {
                console.log('No rows in daily_trends data');
                console.log('data object:', data);
                noDataMsg.style.display = 'block';
                canvas.style.display = 'none';
                // Destroy existing chart if any
                if (state.charts.dailyTrends) {
                    state.charts.dailyTrends.destroy();
                    state.charts.dailyTrends = null;
                }
                return;
            }
            
            console.log(`Creating chart with ${data.rows.length} rows`);
            // We have data, hide no-data message and show canvas
            noDataMsg.style.display = 'none';
            canvas.style.display = 'block';
            
            const days = data.rows.map(row => row.day);
            const totals = data.rows.map(row => row.total_referrals);
            const completed = data.rows.map(row => row.completed || 0);
            
            console.log('Days:', days);
            console.log('Totals:', totals);
            console.log('Completed:', completed);
            
            try {
                state.charts.dailyTrends = new Chart(ctx, {
                    type: 'line',
                    data: {
                        labels: days,
                        datasets: [
                            {
                                label: 'Total Referrals',
                                data: totals,
                                borderColor: '#3498db',
                                backgroundColor: 'rgba(52, 152, 219, 0.1)',
                                tension: 0.4,
                                fill: true
                            },
                            {
                                label: 'Completed',
                                data: completed,
                                borderColor: '#2ecc71',
                                backgroundColor: 'rgba(46, 204, 113, 0.1)',
                                tension: 0.4,
                                fill: true
                            }
                        ]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            legend: { position: 'top' }
                        },
                        scales: {
                            y: {
                                beginAtZero: true,
                                title: {
                                    display: true,
                                    text: 'Count'
                                }
                            }
                        }
                    }
                });
                console.log('Chart created successfully');
            } catch (error) {
                console.error('Error creating chart:', error);
            }
        }

        function updateDataTable() {
            const data = state.data.monthly_summary || { rows: [] };
            const tableBody = document.querySelector('#monthlyTable tbody');
            
            if (!data.rows || data.rows.length === 0) {
                tableBody.innerHTML = '<tr><td colspan="5">No data available</td></tr>';
                return;
            }

            let html = '';
            data.rows.forEach(row => {
                html += `<tr>
                    <td>${row.month || ''}</td>
                    <td>${row.total_referrals || 0}</td>
                    <td>${row.unique_seekers || 0}</td>
                    <td>${row.unique_senders || 0}</td>
                    <td>${row.completed || 0}</td>
                </tr>`;
            });
            
            tableBody.innerHTML = html;
        }

        function updateStatsPanel() {
            try {
                const dailyData = state.data.daily_trends || { rows: [] };
                const monthlyData = state.data.monthly_summary || { rows: [] };
                
                let totalReferrals = 0;
                let completedReferrals = 0;
                
                if (dailyData.rows && dailyData.rows.length > 0) {
                    dailyData.rows.forEach(row => {
                        totalReferrals += Number(row.total_referrals) || 0;
                        completedReferrals += Number(row.completed) || 0;
                    });
                }
                
                const completionRate = totalReferrals > 0 ? (completedReferrals / totalReferrals * 100).toFixed(1) : 0;
                
                let uniqueSeekers = 0;
                if (monthlyData.rows && monthlyData.rows.length > 0) {
                    monthlyData.rows.forEach(row => {
                        uniqueSeekers += Number(row.unique_seekers) || 0;
                    });
                }
                
                document.getElementById('total-referrals').textContent = totalReferrals.toLocaleString();
                document.getElementById('completed-referrals').textContent = completedReferrals.toLocaleString();
                document.getElementById('completion-rate').textContent = `${completionRate}%`;
                document.getElementById('unique-seekers').textContent = uniqueSeekers.toLocaleString();
                
            } catch (error) {
                console.error('Error updating stats panel:', error);
            }
        }

        function displayImmutableParams() {
            const container = document.getElementById('immutable-params');
            const config = window.__REPORT_CONFIG__ || {};
            const params = config.params || {};
            const immutableParams = config.immutable_params || [];
            
            if (immutableParams.length === 0) {
                container.style.display = 'none';
                return;
            }

            let html = '<h4>Security Context (Immutable Parameters)</h4><div class="param-grid">';
            immutableParams.forEach(paramName => {
                html += `
                    <div class="param-item">
                        <span class="param-label">${paramName}</span>
                        <span class="param-value">${params[paramName] || '(not set)'}</span>
                    </div>
                `;
            });
            html += '</div>';
            container.innerHTML = html;
        }

        function destroyCharts() {
            Object.values(state.charts).forEach(chart => {
                if (chart && chart.destroy) {
                    chart.destroy();
                }
            });
            
            // Reset chart references
            state.charts = {
                dailyTrends: null,
                statusDistribution: null,
                contactMethod: null,
                monthlySummary: null
            };
        }

        function showError(message) {
            const errorEl = document.getElementById('error-message');
            errorEl.textContent = message;
            errorEl.style.display = 'block';
            setTimeout(() => {
                errorEl.style.display = 'none';
            }, 5000);
        }

        // Wait for thick client to be ready before initializing dashboard
        function whenThickClientReady() {
            return new Promise((resolve) => {
                // If thick client is already ready, resolve immediately
                if (window.ReportApp && window.ReportApp.refresh) {
                    console.log('Thick client already ready');
                    resolve();
                    return;
                }
                // Otherwise wait for the data-ready event
                const listener = (event) => {
                    console.log('Thick client data-ready event received', event);
                    window.removeEventListener('report:data-ready', listener);
                    resolve();
                };
                window.addEventListener('report:data-ready', listener);
                // Fallback: If event never fires, check every 100ms for up to 10 seconds
                let attempts = 0;
                const interval = setInterval(() => {
                    if (window.ReportApp && window.ReportApp.refresh) {
                        clearInterval(interval);
                        window.removeEventListener('report:data-ready', listener);
                        resolve();
                    }
                    attempts++;
                    if (attempts > 100) { // 10 seconds
                        clearInterval(interval);
                        window.removeEventListener('report:data-ready', listener);
                        console.warn('Timeout waiting for thick client');
                        resolve(); // still try to proceed
                    }
                }, 100);
            });
        }

        // Initialize dashboard when DOM is ready
        document.addEventListener('DOMContentLoaded', async function() {
            console.log('DOM ready, waiting for thick client...');
            
            await whenThickClientReady();
            
            console.log('Thick client ready, starting dashboard...');
            // Start the dashboard
            initDashboard();
        });
    </script>
